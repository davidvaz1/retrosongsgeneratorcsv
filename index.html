const fetch = require('node-fetch'); exports.handler = async (event) => { const clientId = event.headers['x-client-id']; const clientSecret = event.headers['x-client-secret']; const offset = event.headers['x-offset'] || 0; try { // 1. Authenticate server-side const authRes = await fetch('https://accounts.spotify.com/api/token', { method: 'POST', headers: { 'Authorization': 'Basic ' + Buffer.from(clientId + ':' + clientSecret).toString('base64'), 'Content-Type': 'application/x-www-form-urlencoded' }, body: 'grant_type=client_credentials' }); if (!authRes.ok) return { statusCode: 401, body: JSON.stringify({ message: "Invalid Spotify Credentials" }) }; const { access_token } = await authRes.json(); // 2. Search using the Feb 2026 limit of 10 const searchUrl = `https://api.spotify.com/v1/search?q=year:2026&type=track&limit=10&offset=${offset}`; const searchRes = await fetch(searchUrl, { headers: { 'Authorization': `Bearer ${access_token}` } }); const data = await searchRes.json(); return { statusCode: 200, body: JSON.stringify(data) }; } catch (error) { return { statusCode: 500, body: JSON.stringify({ message: error.message }) }; } };
